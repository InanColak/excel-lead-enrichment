    ```python
    from datetime import datetime
    from sqlalchemy import func

    async def list_jobs(
        db: AsyncSession,
        user_id: uuid.UUID,
        limit: int = 20,
        offset: int = 0,
        status: str | None = None,
        created_after: datetime | None = None,
        created_before: datetime | None = None,
    ) -> tuple[list[Job], int]:
        query = select(Job).where(Job.user_id == user_id)
        count_query = select(func.count()).select_from(Job).where(Job.user_id == user_id)

        if status:
            query = query.where(Job.status == status)
            count_query = count_query.where(Job.status == status)
        if created_after:
            query = query.where(Job.created_at >= created_after)
            count_query = count_query.where(Job.created_at >= created_after)
        if created_before:
            query = query.where(Job.created_at <= created_before)
            count_query = count_query.where(Job.created_at <= created_before)

        total = (await db.execute(count_query)).scalar() or 0
        jobs = (await db.execute(
            query.order_by(Job.created_at.desc()).offset(offset).limit(limit)
        )).scalars().all()

        return jobs, total
    ```

    **2. Add `get_user_stats` to `app/jobs/service.py`:**
    ```python
    async def get_user_stats(
        db: AsyncSession,
        user_id: uuid.UUID,
        since: datetime | None = None,
        until: datetime | None = None,
    ) -> dict:
        from sqlalchemy import func, coalesce

        base_filter = [Job.user_id == user_id]
        if since:
            base_filter.append(Job.created_at >= since)
        if until:
            base_filter.append(Job.created_at <= until)

        # Aggregation query
        agg_query = select(
            func.count(Job.id).label("total_jobs"),
            func.coalesce(func.sum(Job.api_calls), 0).label("total_api_calls"),
            func.coalesce(func.sum(Job.cache_hits), 0).label("total_cache_hits"),
            func.coalesce(func.sum(Job.webhook_callbacks_received), 0).label("total_webhook_callbacks"),
            func.coalesce(func.sum(Job.webhook_timeouts), 0).label("total_webhook_timeouts"),
        ).where(*base_filter)

        result = await db.execute(agg_query)
        row = result.one()

        total_api_calls = row.total_api_calls
        total_cache_hits = row.total_cache_hits
        total_lookups = total_api_calls + total_cache_hits
        cache_hit_rate = round((total_cache_hits / total_lookups * 100), 1) if total_lookups > 0 else 0.0

        # Jobs by status breakdown
        status_query = select(
            Job.status, func.count(Job.id)
        ).where(*base_filter).group_by(Job.status)
        status_result = await db.execute(status_query)
        jobs_by_status = {status: count for status, count in status_result.all()}

        return {
            "total_jobs": row.total_jobs,
            "total_api_calls": total_api_calls,
            "total_cache_hits": total_cache_hits,
            "cache_hit_rate_percent": cache_hit_rate,
            "total_webhook_callbacks": row.total_webhook_callbacks,
            "total_webhook_timeouts": row.total_webhook_timeouts,
            "jobs_by_status": jobs_by_status,
        }
    ```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-03 | Information Disclosure | GET /api/v1/jobs | mitigate | All queries filter by `Job.user_id == user.id` from JWT. Reuses existing `get_job_by_id` ownership check for single-job endpoints. `list_jobs` adds user_id to WHERE clause. |
| T-04-04 | Information Disclosure | GET /api/v1/jobs/{id}/download | mitigate | `get_job_by_id` verifies ownership before serving file. Path traversal impossible since `output_file_path` is server-generated (never from user input). `FileResponse` serves from absolute path. |

| T-04-07 | Information Disclosure | output_file_path in response | mitigate | `Field(exclude=True)` on output_file_path prevents server filesystem paths from leaking to API clients. `has_output` boolean exposed instead. |
